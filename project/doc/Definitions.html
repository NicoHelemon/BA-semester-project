Dioids

Section Dioids.

Class Dioid {A: Type}(plus : A → A → A)(mult : A → A → A)(zero : A)(one : A) : Prop := {
plus_assoc : ∀ x y z, plus x (plus y z) = plus (plus x y) z;
mult_assoc : ∀ x y z, mult x (mult y z) = mult (mult x y) z;
plus_com : ∀ x y, plus x y = plus y x;
plus_neutral : ∀ x, plus zero x = x;
mult_neutral_l : ∀ x, mult one x = x;
mult_neutral_r : ∀ x, mult x one = x;
mult_abs_l : ∀ x, mult zero x = zero;
mult_abs_r : ∀ x, mult x zero = zero;
distrib_l : ∀ x y z, mult x (plus y z) = plus (mult x y) (mult x z);
distrib_r : ∀ x y z, mult (plus x y) z = plus (mult x z) (mult y z);
idempotent: ∀ x, plus x x = x
}.

End Dioids.

Non decreasing functions


Definition non_decreasing (f: R → Rbar) := ∀ s t, (s ≥ 0 ∧ t ≥ 0) → (s < t → Rbar_le (f(s)) (f(t))).
Definition non_decreasing_2 (f: R → Rbar) := ∀ s t, (s ≥ 0 ∧ t ≥ 0) → (s ≤ t → Rbar_le (f(s)) (f(t))).

Theorem non_decr_eq : ∀ f, non_decreasing f ↔ non_decreasing_2 f.

Lemma ex_or_distrib {T: Type} : ∀ (P P' : T → Prop), (∃ s : T, (P s ∨ P' s)) ↔ (∃ s : T, P s) ∨ (∃ s : T, P' s).

Theorem non_decr_plus `(P: non_decreasing p) (v : R) : non_decreasing (fun t ⇒ Rbar_plus (p t) v).

Cumulative functions

Definitions


Class NotMInf (f: R → Rbar) := {
  not_inf: ∀ x, x ≥ 0 → (m_infty) ≠ (f x)
}.

Class Positive (f: R → Rbar) := {
  pos_to_pos : ∀ x, x ≥ 0 → Rbar_le 0 (f x)
}.

Class NonDecreasingPositive (f: R → Rbar) := {
  positive :> Positive f;
  non_decr: non_decreasing f
}.

Class Cumulative (f: R → Rbar) := {
  non_decr_pos :> NonDecreasingPositive f;
  left_cont: ∀ x, x ≥ 0 → filterlim f (at_right x) (Rbar_locally (f x));
  start_0 : f(0) = 0;
}.

  Definition continuous (f: R → Rbar) := ∀ x: R, x ≥ 0 → filterlim f (locally x) (Rbar_locally (f x)).

Delta function


  Definition delta (T: Rbar) : R → Rbar := fun x ⇒ match Rle_lt_dec x T with

    | right _ ⇒ p_infty
    end.

    Qed.

Subadditive closure


  Fixpoint conv_pow (f: R → Rbar) (n: nat) :=
  match n with
  | O ⇒ neutral_conv
  | S n' ⇒ conv f (conv_pow f n')

Definition subadd_clos (alpha: R → Rbar): (R → Rbar) :=
(fun x: R ⇒ Inf_seq (fun n: nat ⇒ conv_pow alpha n x)).

Server


Section Server.

Class Server (S : (R → Rbar) → (R → Rbar) → Prop) := {
  subset_c_c: ∀ a d : R → Rbar, S a d → Cumulative a ∧ Cumulative d;
  completeness: ∀ a : R → Rbar, Cumulative a → ∃ d, S a d;
  causality: ∀ a d t, S a d → t ≥ 0 → Rbar_le (d(t)) (a(t))
}.


Theorem finite_server_t `(S: Server s): ∀ a d t, t ≥ 0 → s a d → is_finite (a t) → is_finite (d t).

Theorem finite_server `(S: Server s) : ∀ a d, s a d → is_finite_f a → is_finite_f d.

Class MultiInput (S : list (R → Rbar) → list (R → Rbar) → Prop) (n: nat) := {
  same_length: ∀ (A D : list (R → Rbar)), S A D → length A = n ∨ length D = n;
  cumul_lists: ∀ (A D : list (R → Rbar)) (a d : R → Rbar), S A D → (List.In a A → Cumulative a ∨ List.In d D → Cumulative d);
  completeness_sched: ∀ (A : list (R → Rbar)), (∀ a, List.In a A → Cumulative a) → ∃ D, S A D;
  causality_sched: ∀ (A D : list (R → Rbar)) (i: nat) (t: R), S A D → lt i n → List.nth i A (delta 0) t ≥ List.nth i D (delta 0) t
}.


Server concatenation


Definition server_concat `(S: Server s) `(T: Server t) := fun a c ⇒ ∃ b, (s a b) ∧ (t b c).

Global Instance ServerConcat `(S: Server s) `(T: Server t): Server (server_concat S T).
Qed.
