Finite infimum and supremum of real sets


Section Reals.

Definitions


  Definition ex_sup (S: Ensemble R) := ∃ l, is_lub S l.

  Definition is_lower_bound (E:R → Prop) (m:R) := ∀ x:R, E x → x ≥ m.

  Definition is_glb (E:R → Prop) (m:R) :=
    is_lower_bound E m ∧ (∀ b:R, is_lower_bound E b → m ≥ b).

  Definition ex_inf (S: Ensemble R) := ∃ l, is_glb S l.

  Definition sup_ex (S: Ensemble R) := { m : R | is_lub S m}.

    Theorem is_sup_ex (S: Ensemble R) (s: R) :
    is_lub S s → sup_ex S.

    Definition sup (S: Ensemble R) (ex: sup_ex S) := proj1_sig(ex).

    Axiom ex_sup_cond : ∀ S, R_bounded S → S ≠ Empty_set R → sup_ex S.

    Theorem sup_unique (S: Ensemble R) (p: sup_ex S) (s: R) :
    is_lub S s → sup S p = s.

    Theorem sup_is_sup (S: Ensemble R) (ex: sup_ex S) :
    is_lub S (sup S ex).

Relations between supremum and infimum


  Theorem is_lub_opp: ∀ E l,
  is_glb E l ↔ is_lub (set_opp E) (- l).

  Theorem is_glb_opp: ∀ E l,
  is_lub E l ↔ is_glb (set_opp E) (- l).

  Theorem ex_inf_opp : ∀ S, ex_sup S ↔ ex_inf (set_opp S).

  Theorem ex_sup_opp : ∀ S, ex_inf S ↔ ex_sup (set_opp S).

  Theorem sup_non_empty : ∀ S l, is_lub S l → ∃ s, S s.

  Theorem inf_non_empty: ∀ S l, is_glb S l → ∃ s, S s.

End Reals.
