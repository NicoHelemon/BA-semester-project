Arrival curves


Definition is_arrival_curve (a: R → Rbar) `(alpha: NonDecreasingPositive al) :=
∀ x, Cumulative a → x ≥ 0 → Rbar_le (a(x)) (conv al a x).

Definition is_arrival_curve_2 (a: R → Rbar) `(alpha: NonDecreasingPositive al) :=
Cumulative a → ∀ s t, s ≥ 0 → t ≥ 0 → s ≤ t → Rbar_le (Rbar_minus (a t) (a s)) (al(t - s)).

Theorem arr_cur_eq `(alpha: NonDecreasingPositive al) : ∀ a, is_finite_f a → (is_arrival_curve a alpha ↔ is_arrival_curve_2 a alpha).

Theorem arr_cur_eq_start_0 `(A: Cumulative a)`(alpha: NonDecreasingPositive al) : al 0 = 0 → (is_arrival_curve a alpha ↔ ∀ x, x ≥ 0 → a(x) = conv al a x).

Theorem subadd_clos_is_arr_curve `(A: Cumulative a) `(alpha: NonDecreasingPositive al) : is_arrival_curve a alpha → is_arrival_curve a (SubaddClos alpha).

Minimal service curve


Definition is_minimal_service `(S: Server s) `(Beta: NonDecreasingPositive beta) := ∀ a d t, s a d → t ≥ 0 → Rbar_le (conv a beta t) (d t).

Definition is_minimal_service_2 `(S: Server s) `(Beta: NonDecreasingPositive beta) := ∀ a d t, s a d → t ≥ 0 → (∃ s, s ≥ 0 ∧ s ≤ t ∧ Rbar_le (Rbar_plus (a (t - s)) (beta s)) (d t)).

Theorem min_ser_eq `(S: Server s) `(Beta: NonDecreasingPositive beta) : continuous beta → (is_minimal_service S Beta ↔ is_minimal_service_2 S Beta).
Theorem is_minimal_service_0 `(S: Server) : is_minimal_service S (zero_f_non_decr).


Theorem minimal_service_plus `(S: Server s) `(P: non_decreasing p) : (∀ a d t, s a d → t ≥ 0 → Rbar_le (conv a p t) (d t)) → is_minimal_service S (NonDecrFPlus P).

Backlog


Definition backlog_t (a d: R → Rbar)(t: R) := Rbar_minus (a t) (d t).

Definition backlog (a d: R → Rbar) := Rbar_lub(Im R Rbar (fun x ⇒ x ≥ 0) (backlog_t a d)).

Definition is_backlogged (a d: R → Rbar)(t: R) := t ≥ 0 → Rbar_lt 0 (backlog_t a d t).

Theorem backlog_bound `(S: Server s) `(alpha: NonDecreasingPositive al) `(Beta: NonDecreasingPositive beta):
∀ a d, is_finite_f a → s a d → (is_arrival_curve a alpha) → (is_minimal_service S Beta) → Rbar_le (backlog a d) (backlog al beta).

Definition ex_begin_backlogged_period (a d: R → Rbar)(t: R) := ∃ l, 0 ≤ l ≤ t ∧ ∀ s, l < s ≤ t → is_backlogged a d s.

Definition ex_end_backlogged_period (a d: R → Rbar)(t: R) := ∃ h, t ≤ h ∧ ∀ s, t ≤ s < h → is_backlogged a d s.

Delay


Definition delay_t (a d: R → Rbar) (t: R): Rbar := Glb_Rbar(fun delta: R ⇒ Rbar_le (a t) (d(t + delta))).

Definition delay (a d: R → Rbar): Rbar := Rbar_lub(Im R Rbar (fun x ⇒ x ≥ 0) (delay_t a d)).

Theorem lt_delay_t : ∀ (a d: R → Rbar) (t tao: R), Rbar_lt tao (delay_t a d t) → Rbar_lt (d (t + tao) ) (a t).
