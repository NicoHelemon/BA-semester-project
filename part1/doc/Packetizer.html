Packetizer

This section defines a packetizer and proves theorems about it. After that the concatenation between a bit by bit system and a scheduler is analysed.


Packet below


Definition packet_below (l: nat → nat) (b: R): Ensemble nat :=
match Rle_dec 0 b with
|left _ ⇒ (fun n: nat ⇒ Rle (INR (nat_sum l n)) b)
|right _ ⇒ Singleton nat O
end.

Theorem packet_below_incl: ∀ (l: nat → nat) (c: l O = O) (b1 b2 : R), b1 ≤ b2 → Included nat (packet_below l b1) (packet_below l b2).

Theorem packet_below_bounded : ∀ l b, nat_bounded (packet_below l b).

Theorem packet_below_non_empty: ∀ l b, l O = O → packet_below l b ≠ Empty_set nat.

Definition last_packet (l: nat → nat) (z: l O = O) (b: R):=
nat_max (packet_below l b) (packet_below_bounded l b) (packet_below_non_empty l b z).

Theorem last_packet_0 : ∀ l z (sp: strict_pos l), last_packet l z 0 = O.

Theorem last_packet_le : ∀ l b1 b2 (z: l O = O), b1 ≤ b2 → le (last_packet l z b1) (last_packet l z b2).

Theorem last_packet_bounds : ∀ l z b n (p: b ≥ 0), INR(nat_sum l n) ≤ b < INR(nat_sum l (S n)) ↔ last_packet l z b = n.

Cumul below


Definition cumul_below (l: nat → nat) (b: R) := fun r ⇒ ∃ n, packet_below l b n ∧ r = INR (nat_sum l n).

Theorem cumul_below_non_empty: ∀ l b, l O = O → cumul_below l b ≠ Empty_set R.

Theorem cumul_below_bounded : ∀ l b, set_bounded_above (cumul_below l b) Rle Rle_order.

Definition last_cumul (l: nat → nat) (z: l O = O) (b: R) : R:=
R_max (cumul_below l b) (cumul_below_bounded l b) (cumul_below_non_empty l b z).

Theorem cumul_below_incl: ∀ (l: nat → nat) (c: l O = O) (b1 b2 : R), b1 ≤ b2 → Included R (cumul_below l b1) (cumul_below l b2).

Theorem last_cumul_le : ∀ l b1 b2 (z: l O = O), b1 ≤ b2 → Rle (last_cumul l z b1) (last_cumul l z b2).

Relation between last packet and last cumul


Theorem last_packet_in_cumul (l: nat → nat) (z: l O = O) (b: R) : cumul_below l b (INR (nat_sum l (last_packet l z b))).

Theorem last_cumul_packet (l: nat → nat) (z: l O = O) (b: R) :
last_cumul l z b = INR(nat_sum l (last_packet l z b)).

Packetization



Class LengthSequence (l: nat → nat) := {
  str_pos : strict_pos l;
  im_0: l O = O;
  ex_max: nat_seq_ex_max l
}.

Definition lmax `(ls: LengthSequence l) := nat_seq_max l ex_max.

Definition packetization (l: nat → nat) (ls: LengthSequence l) (v: Rbar): Rbar :=
  match v with
  |Finite x ⇒ last_cumul l im_0 x
  |p_infty ⇒ p_infty
  |m_infty ⇒ 0
  end.

Theorem packetization_0 (l: nat → nat) (ls: LengthSequence l): packetization l ls 0 = 0.

Theorem packetization_eq : ∀ l ls (v: R) (p: v ≥ 0) n, packetization l ls v = INR(nat_sum l n) ↔ INR(nat_sum l n) ≤ v < INR(nat_sum l (S n)).

Theorem packetization_is_sum: ∀ l ls (v: R), ∃ n, packetization l ls v = INR(nat_sum l n).

Theorem packetization_bounds: ∀ l ls (v: R) (p: v ≥ 0), v - (INR(lmax ls)) < packetization l ls v ≤ v.

Theorem packetization_idempotent (a: R → Rbar) (l: nat → nat) ls v : packetization l ls (packetization l ls v) = packetization l ls v.

Definition packetization_isotone_2 : ∀ (a: R → Rbar) (l: nat → nat) ls (x: Rbar) (y: Rbar), Rbar_le x y → Rbar_le (packetization l ls x) (packetization l ls y).
Qed.

Packetizer




Definition packetizer (a: R → Rbar) (l: nat → nat) ls : R → Rbar := fun t ⇒ packetization l ls (a t).

Definition packetized `(ls : LengthSequence l) (a: R → Rbar) : Prop := packetizer a l ls = a.

Theorem packetizer_cumul `(A: Cumulative a) : ∀ l ls, Cumulative (packetizer a l ls).

Definition packetizer_serv `(ls: LengthSequence l) := fun a d ⇒ Cumulative a ∧ d = packetizer a l ls.

Global Instance Packetizer `(ls: LengthSequence l) : Server (packetizer_serv ls).

Qed.


Theorem backlog_packetizer : ∀ `(A: Cumulative a) (l: nat → nat) (ls: LengthSequence l),
Rbar_le (backlog a (packetizer a l ls)) (INR (lmax ls)).

Theorem packetizer_finite_t `(A: Cumulative a): ∀ l ls t, t ≥ 0 → is_finite (a t) ↔ is_finite (packetizer a l ls t).

Theorem packetizer_finite `(A: Cumulative a): ∀ l ls, is_finite_f a ↔ is_finite_f (packetizer a l ls ).

Global Instance NonDecrPlusLmax `(P: NonDecreasingPositive p) `(ls: LengthSequence l): NonDecreasingPositive (fun t ⇒ Rbar_plus (p t) (INR(lmax ls))).
Qed.

Theorem packetizer_arrival_curve `(A: Cumulative a) `(alpha: NonDecreasingPositive al) `(ls: LengthSequence l) :
is_finite_f a → is_arrival_curve a alpha → is_arrival_curve (packetizer a l ls) (NonDecrPlusLmax alpha ls).

Arrival times


Definition arrival_time `(ls: LengthSequence l) (a: R → Rbar) := (fun n ⇒ Glb_Rbar (fun t ⇒ t ≥ 0 ∧ (packetizer a l ls) t = INR(nat_sum l n))).

Theorem arrival_time_0 `(ls : LengthSequence l) : ∀ a (A: Cumulative a), arrival_time ls a 0 = 0.

Theorem arrival_time_bounded `(ls : LengthSequence l) : ∀ a (A: Cumulative a) (t: R) n, set_is_bound_above (Im R Rbar (fun x ⇒ x ≥ 0) (packetizer a l ls)) Rbar_le Rbar_le_order (INR(nat_sum l n)) → (arrival_time ls a (S n)) = p_infty.

Theorem arrival_time_partition `(ls : LengthSequence l) : ∀ a (A: Cumulative a) (t: R), ∃ i, Rbar_le t (arrival_time ls a i).

Combined systems


Definition combined_system `(S: Server s) `(ls: LengthSequence l) := server_concat S (Packetizer ls).

Global Instance CombinedSystem `(S: Server s) `(ls: LengthSequence l) : Server (combined_system S ls).
Qed.


Theorem combined_backlog `(S: Server s) `(ls : LengthSequence l) : ∀ a d d',s a d → packetizer_serv ls d d' → Rbar_le(backlog a d) (backlog a d') ∧ Rbar_le (backlog a d') (Rbar_plus (backlog a d) (INR(lmax ls))).

Theorem combined_service `(S: Server s) `(ls : LengthSequence l) `(Beta: NonDecreasingPositive beta): is_minimal_service S Beta → is_minimal_service (CombinedSystem S ls) (NonDecrFPlus (non_decr_plus non_decr (-INR (lmax ls)))).

Theorem combined_delay `(S: Server s) `(ls: LengthSequence l) : ∀ a d d' (p: packetized ls a), s a d → (combined_system S ls) d d' → delay a d = delay a d'.
